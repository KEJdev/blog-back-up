N,M = map(int, input().split())
board, temp = [],[]
for _ in range(N) :
    board.append(input())

for i in range(N-7):
    for k in range(M-7):
        b,w = 0,0 
        for i_a in range(i,i+8):
            for k_a in range(k,k+8):
                if (i_a+k_a)%2 == 0:
                    if board[i_a][k_a] != 'W':
                        b+=1
                    else:
                        w+=1
                else:
                    if board[i_a][k_a] != 'B':
                        b+=1
                    else:
                        w+=1
        temp.append(min(b,w))
print(min(temp))
