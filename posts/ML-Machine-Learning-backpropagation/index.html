class Affine:
  def __init__(self, w,b):
    self.w = w
    self.b = b

  def forward(self,x):
    out = np.dot(x, self.w)+self.b
    return out

  def backward(self, x, dy):
    dx = np.dot(dy, self.w.T)
    dw = np.dot(x.T, dy)
    db = np.sum(dy,axis=0)
    return dx,dw,db
