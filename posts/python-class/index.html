클래스(class))

class gun():

    # 클래스가 실체화 될때, 바로 작동하는 메소드(함수)
    def __init__(self): # 총을 만든다. (실체화)

        # 총알을 0으로 설정 
        self.bullet = 0

    # class에는 꼭 self가 들어간다. 
    def charge(self, num):#충전하는 기능
        self.bullet=num

    def shoot(self, num): #쏘는 기능
        for i in range(num):
            if self.bullet>0:
                print('탕 !')
                self.bullet-= 1
            elif self.bullet == 0:
                print('총알이 없습니다.')
            break

class gun():

    # 클래스가 실체화 될때, 바로 작동하는 메소드(함수)
    def __init__(self): # 총을 만든다. (실체화)

        # 총알을 0으로 설정 
        self.bullet = 0

    # class에는 꼭 self가 들어간다. 
    def charge(self, num):#충전하는 기능
        self.bullet=num

    def shoot(self, num): #쏘는 기능
        for i in range(num):
            if self.bullet>0:
                print('탕 !')
                self.bullet-= 1
            elif self.bullet == 0:
                print('총알이 없습니다.')
            break

    def print(self): #출력하는 기능
        print('{}발 남았습니다.'.format(self.bullet))
